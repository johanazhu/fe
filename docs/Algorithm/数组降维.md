# 数组降维

```javascript
var arr = [2, 3, [2, 2],
    [3, 'f', ['w', 3]], { "name": 'Tom' }
];
function deepFlatten(arr) {
    var result = [];
    arr.forEach(function(val, index) {
        if(Array.isArray(val)) {
            val.forEach(arguments.callee)
        } else {
            result.push(val)
        }
    })
    return result
}
console.log(deepFlatten(arr))
```
