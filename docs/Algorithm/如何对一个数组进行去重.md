# 如何对一个数组进行去重

```javascript
var arr = [1, 2, 50, 1, 60, 20,null];
var result = arr.indexOf(60);
console.log(result)

function unique(arr){
    var newArr = [];
    var item;
    for(var i =0; i< arr.length; i++ ){
        item = arr[i];
        if(newArr.indexOf(item) === -1){
            newArr.push(item)
        }
    }
    return newArr;
}

var arr1 = unique(arr)
console.log(arr1)
```
