```react

   this.state = {
       a: 0;
   }
   componentDidMount() {
       this.setState({ a: 1 })
       setTimeout(() => {
           	this.setState({a: 2})
       }, 0)
       new Promise((resolve) => {
           resolve(this.setState({ a: 3}))
       }).then(() => { this.setState({a:4}) })
   }
}
```
